1: git init
2: git add . or git add <file name>
3: git commit -m " <message> "
4: git push origin <branch>

5: git clone <SSH Key> - get repository from remote

6: git config --global user.name -to show user 
7: git config --global user.name <name> - to set new user name
8: git config --global user.email - to show email
9: git config --global user.email <email> -to set new email 

10: git remote add origin <SSH Key> - to add repository from local to remote

11: git branch - to check branches 
12: git chechout <branch name> - to change branch
13: git chechout -M <branch name> -to create new branch
